#### Yearly check to see if event should fire for Chiselgrip or Thorium Point owners
yearly_global_pulse = { 
	on_actions = { thorium_base_check } 
}

thorium_base_check = {
	trigger = {
		if = {
			limit = {
				OR = {
					title:c_drognur.holder = { 	# Thorium Point
						NOT = { 
							religion = religion:fire_group
							religion = religion:doombringer
							religion = religion:deathbringer
							religion = religion:shathgral
						} 
					}
					title:c_chiselgrip.holder = { 
						NOT = { 
							religion = religion:fire_group
							religion = religion:doombringer
							religion = religion:deathbringer
							religion = religion:shathgral
						} 
					}
				}
			}
		}
	}
	effect = {
		trigger_event = { on_action = set_thorium_base_modifier months = { 2 4 } }
	}
}
	
set_thorium_base_modifier = {
	effect = {
		capital_province = {
			add_province_modifier = {
				modifier = wc_thorium_base_modifier
			}
		}
	}
}
